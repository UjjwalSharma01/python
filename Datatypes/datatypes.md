- String : 'spam', "guido's", b'a\x01c', u'sp\xc4m'
- List : [1, [2, 'three'], 4] 
- Tuple : (1, 'spam', 4, 'U')
- Dictionary : {'name': 'ujjwal','age': 18}
- Set : {'a','b','c'}, set('abc)
- File : open(ujjwal.txt) 
- Boolean : True, False
- None : None
